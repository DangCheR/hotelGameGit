local util = require 'xlua.util'
--这个神奇的函数可以将一个需要回调的函数转换成同步函数
--转换后的函数结束后挂起协程，回调函数被包装成恢复协程
--回调函数会把结果传递给协程，让协程继续执行

local gameobject = CS.UnityEngine.GameObject('MessageBox')
CS.UnityEngine.Object.DontDestroyOnLoad(gameobject)
local messageBox = gameobject:AddComponent(typeof(CS.XLuaTest.MessageBox))

local sync_alert = util.async_to_sync(function(msg, title, cb)
    messageBox:ShowAlertBox(msg, title, cb)  -- ← 注意这里是 : 不是 .
end)

local sync_confirm = util.async_to_sync(function(msg, title, cb)
    messageBox:ShowConfirmBox(msg, title, cb)
end)

-- local sync_alert = util.async_to_sync(messageBox.ShowAlertBox)
-- local sync_confirm = util.async_to_sync(messageBox.ShowConfirmBox) 

-- local sync_alert = util.async_to_sync(CS.XLuaTest.MessageBox.ShowAlertBox)
-- local sync_confirm = util.async_to_sync(CS.XLuaTest.MessageBox.ShowConfirmBox) 

--构造alert和confirm函数
return {
    alert = function(message, title)
		 sync_alert(message, title)
    end;
	
	confirm = function(message, title)
		local ret = sync_confirm(message, title)
		return ret == true
    end;
 }
